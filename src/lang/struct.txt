create_object2(obj Object) -> Object2 {
    Object2 {
        obj: obj
    }
}

Object2 {
    obj Object
}

Object {
    isEnabled bool,
    x float,
    y float,
}

main(arg1 float, arg2 float) -> float {
    obj = Object {
        isEnabled: true,
        x: arg1,
        y: arg2,
    }

    obj2 = create_object2(obj)
    obj2.obj.x
}
